# 函数柯里化

```js
const curry = (func) => {
    let curriedFn;
    return curriedFn = (...args) => {
        if (args.length < func.length) {
            return (...args1) => {
                return curriedFunc(...args.concat(args1));
            }
            return func(...args);
        }
    }
}
```
